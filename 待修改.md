    cve         VARCHAR(50),
    risk        VARCHAR(20),   -- Critical/High/Medium/Low/Info
    host        INET,
    port        VARCHAR(10),
    protocol    VARCHAR(10),
    name        TEXT,
    cvss        NUMERIC(4,1),
    epss        NUMERIC(6,4),
    vpr         NUMERIC(4,1),
    synopsis    TEXT,
    description TEXT,
    solution    TEXT,
    plugin_output TEXT
);

-- NIST Audit
CREATE TABLE audit_scans (
    id          SERIAL PRIMARY KEY,
    name        VARCHAR(200),
    scan_date   DATE,
    uploaded_at TIMESTAMPTZ DEFAULT NOW(),
    total       INT,
    passed      INT,
    failed      INT,
    warning     INT
);

CREATE TABLE audit_results (
    id          SERIAL PRIMARY KEY,
    scan_id     INT REFERENCES audit_scans(id) ON DELETE CASCADE,
    check_name  TEXT,
    status      VARCHAR(20),   -- PASSED/FAILED/WARNING
    description TEXT,
    policy_val  TEXT,
    actual_val  TEXT
);

-- IP 群組
CREATE TABLE ip_groups (
    id          SERIAL PRIMARY KEY,
    name        VARCHAR(100) UNIQUE,
    ips         JSONB,
    created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- 使用者
CREATE TABLE users (
    id          SERIAL PRIMARY KEY,
    username    VARCHAR(100) UNIQUE,
    hashed_pw   VARCHAR(255),
    role        VARCHAR(20) DEFAULT 'viewer',  -- admin/analyst/viewer
    created_at  TIMESTAMPTZ DEFAULT NOW()
);
